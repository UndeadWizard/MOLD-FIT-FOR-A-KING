japan_generate_scripted_heirs = {
	if = {
		limit = {
			ruler = {
				has_variable = house_yamato
			}
		}
		if = {
			limit = {
				game_date >= 1831.7.22
				NOT = { has_variable = heir_komei_spawned }
				ruler ?= {
					has_template = JAP_ninko_yamato
				}
			}
			set_variable = heir_komei_spawned
			create_character = {
				template = JAP_komei_yamato
				on_created = {
					replace_character_roles = { add = { heir } remove_all = yes }
				}
			}
		}
		if = {
			limit = {
				game_date >= 1852.11.3
				NOT = { has_variable = heir_meiji_spawned }
				ruler ?= {
					has_template = JAP_komei_yamato
				}
			}
			set_variable = heir_meiji_spawned
			create_character = {
				template = JAP_meiji_yamato
				on_created = {
					replace_character_roles = { add = { heir } remove_all = yes }
				}
			}
		}
		else_if = {
			limit = {
				game_date >= 1879.8.31
				NOT = { has_variable = heir_taisho_spawned }
				ruler ?= {
					has_template = JAP_meiji_yamato
				}
			}
			set_variable = heir_taisho_spawned
			create_character = {
				template = JAP_taisho_yamato
				on_created = {
					replace_character_roles = { add = { heir } remove_all = yes }
				}
			}
		}
		else_if = {
			limit = {
				game_date >= 1901.4.29
				NOT = { has_variable = heir_showa_spawned }
				ruler ?= {
					has_template = JAP_taisho_yamato
				}
			}
			set_variable = heir_showa_spawned
			create_character = {
				template = JAP_showa_yamato
				on_created = {
					replace_character_roles = { add = { heir } remove_all = yes }
				}
			}
		}
	}
	else_if = {
		limit = {
			ruler = {
				has_variable = house_tokugawa
			}
		}
		if = {
			limit = {
				NOT = { is_template_used = JAP_iesada_tokugawa }
				ruler ?= {
					has_template = JAP_ieyoshi_tokugawa
				}
			}
			create_character = {
				template = JAP_iesada_tokugawa
				on_created = {
					add_character_role = heir
				}
			}
		}
		else_if = {
			limit = {
				NOT = { has_dlc_feature = ep2_content } # Iesada's heir is handled by event in EP2
				game_date >= 1846.7.16
				NOT = { is_template_used = JAP_iemochi_tokugawa }
				ruler ?= {
					has_template = JAP_iesada_tokugawa
				}
			}
			create_character = {
				template = JAP_iemochi_tokugawa
				on_created = {
					add_character_role = heir
				}
			}
		}
		else_if = {
			limit = {
				game_date >= 1863.8.24
				game_date < 1879.8.24
				NOT = { is_template_used = JAP_yoshinobu_tokugawa }
				ruler ?= {
					has_template = JAP_iemochi_tokugawa
				}
			}
			create_character = {
				template = JAP_yoshinobu_tokugawa
				on_created = {
					add_character_role = heir
				}
			}
		}
		else_if = {
			limit = {
				game_date >= 1879.8.24 # Date of majority
				NOT = { is_template_used = JAP_iesato_tokugawa }
				ruler ?= {
					OR = {
						has_template = JAP_iemochi_tokugawa
						has_template = JAP_yoshinobu_tokugawa
					}
				}
			}
			create_character = {
				template = JAP_iesato_tokugawa
				on_created = {
					add_character_role = heir
				}
			}
		}
	}
}

portugal_generate_scripted_heirs = {
	if = {
		limit = {
			game_date >= 1837.9.16
			NOT = {	has_variable = pedro_v_var }
			ruler ?= {
				has_template = POR_maria_ii_de_braganca
			}
		}
		set_variable = pedro_v_var
		create_character = {
			template = POR_pedro_v_de_braganca_template
		}
	}
}

britain_generate_scripted_heirs = {
	if = {
		limit = {
			game_date >= 1841.11.09
			NOT = { has_variable = edward_vii_heir_var }
			ruler ?= {
				has_template = gbr_queen_victoria_template
			}
		}
		set_variable = edward_vii_heir_var
		create_character = {
			template = gbr_albert_edward_template
			heir = yes
		}
	}
	else_if = {
		limit = {
			game_date >= 1865.06.03
			NOT = { has_variable = george_v_heir_var }
			ruler ?= {
				has_template = gbr_albert_edward_template
			}
		}
		set_variable = george_v_heir_var
		create_character = {
			template = gbr_george_v_template
			heir = yes
		}
	}
}

france_generate_scripted_heirs = {
	if = {
		limit = {
			game_date > 1856.3.16
			NOT = { is_template_used = FRA_napoleon_eugene_louis_bonaparte }
			ruler ?= {
				has_template = fra_napoleon_iii_character_template
			}
		}
		country_summon_heir_effect = {
			TEMPLATE = FRA_napoleon_eugene_louis_bonaparte
		}
	}
}
