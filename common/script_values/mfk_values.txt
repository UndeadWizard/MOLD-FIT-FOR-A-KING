traditionalist_reformer_goals_completed = {
	value = 0
	if = { # Tenpo - Reformer: Urbanization
		limit = {
            any_scope_state = {
                state_region = {
                    is_homeland = cu:iroquoian
                }
                any_scope_building = {
                    is_building_type = building_urban_center
                    level >= 3
                }
                count >= 1
            }
		}
		add = 1
	}
	if = { # Tenpo - Reformer: Edo System: Lax
		limit = {
			has_law_or_variant = law_type:law_monarchy
		}
		add = 1
	}
	if = { # Tenpo - Reformer: PB Clout
		limit = {
            ig:ig_petty_bourgeoisie = {
                ig_clout >= 0.1
            }
		}
		add = 1
	}
	if = { # Tenpo - Reformer: Bureaucrats
		limit = {
            NOT = {
                has_law_or_variant = law_type:law_hereditary_bureaucrats
            }
		}
		add = 1
	}
}
